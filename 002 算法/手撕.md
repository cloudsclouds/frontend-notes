## 1. 发布订阅模式

```js
class EventEmitter {
    constructor() {
        this.events = {};
    }

    // 订阅事件：把回调函数存起来
    on(eventName, callback) {
        if (!this.events[eventName]) {
            this.events[eventName] = [];
        }
        this.events[eventName].push(callback);
    }

    // 触发事件：把这个事件下的所有回调都执行一遍
    emit(eventName, ...args) {
        const subscribers = this.events[eventName];
        if (subscribers) {
            subscribers.forEach((item) => item(...args));
        }
    }

    // 取消订阅：把某个回调从数组里删掉
    off(eventName, callback) {
        const subscribers = this.events[eventName];
        if (subscribers) {
            this.events[eventName] = subscribers.filter((item) => item !== callback);
        }
    }

    // 只触发一次：触发后自动取消订阅
    once(eventName, callback) {
        const fn = (...args) => {
            callback(...args);
            this.off(eventName, fn);
        };
        this.on(eventName, fn);
    }
}

const emitter = new EventEmitter();
function handleMessage(message) {
    console.log('收到消息：' + message);
}
emitter.on('message', handleMessage);
emitter.emit('message', 'hello');
emitter.emit('message', 'world');
emitter.off('message', handleMessage);

function handleLogin(user) {
    console.log('用户登录：' + user);
}
emitter.once('login', handleLogin);
emitter.emit('login', 'Alice');
emitter.emit('login', 'Bob');
```

```js
const nums = [1, 6, 3, 7, 0, 4, 5];
const n = nums.length
for (let i = n - 1; i >= 0; i--) {
    const randomIndex = Math.floor(Math.random()*(i + 1));
    [nums[i], nums[randomIndex]] = [nums[randomIndex], nums[i]];
}
```

## 7. 防抖节流
```js
// 防抖：最后一次触发后，等一段时间再执行
function debounce(fn, delay) {
    let timer = null;

    return function (...args) {
        if (timer) clearTimeout(timer);

        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

const debouncedLog = debounce((num) => {
    console.log('防抖执行：' + num);
}, 200);

debouncedLog(1);
debouncedLog(2);
debouncedLog(3);

// 节流：在规定时间内，只执行第一次，后面的先忽略
function throttle(fn, interval) {
    let lastTime = 0;

    return function (...args) {
        const now = Date.now();

        if (now - lastTime >= interval) {
            fn.apply(this, args);
            lastTime = now;
        }
    };
}

const throttledLog = throttle((num) => {
    console.log('节流执行：' + num);
}, 200);

throttledLog(1);
throttledLog(2);
throttledLog(3);
```

## 8. promise all allSettled race 
```js
Promise.all = function(promises) {
    return new Promise((resolve, reject) => {
        const res = [];
        let count = 0;

        if (promises.length === 0) return resolve([]);

        promises.forEach((p, i) => {
            Promise.resolve(p)
            .then(v => {
                res[i] = value;
                count++;
                if (count === promises.length) {
                    resolve(res);
                }
            })
            .catch(e => reject(e))
        })
    })
}

Promise.allSettled = function(promises) {
    return new Promise((resolve) => {
        const res = [];
        let count = 0;

        if (promises.length === 0) return resolve([]);

        promises.forEach((p, i) => {
            Promise.resolve(p).then(
                (value) => {
                    res[i] = { status: 'fulfilled', value };
                },
                (reason) => {
                    res[i] = { status: 'rejected', reason };
                }
            ).finally(() => {
                count++;
                if (count === promises.length) {
                    resolve(res);
                }
            });
        });
    });
}

Promise.race = function(promises) {
   return new Promise((resolve, reject) => {
        if (promises.length === 0) return;

        promises.forEach((p) => {
            Promise.resolve(p).then(resolve, reject);
        });
    });
}

const p1 = Promise.resolve(1);
const p2 = 2;
const p3 = new Promise((resolve => {
    setTimeout(()=> {
        resolve(3)
    }, 1000);
}))

Promise.all([p1, p2, p3]).then(console.log);
Promise.allSettled([p1, p2, p3]).then(console.log);
Promise.race([p1, p2, p3]).then(console.log);

```
