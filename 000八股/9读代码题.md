## this 指向

```JavaScript
var name = 'window';
const obj = {
  name: 'obj',
  sayName() { console.log(this.name); },
  sayArrow: () => console.log(this.name)
};
obj.sayName();
obj.sayArrow();
const say = obj.sayName;
say();
// obj → window → window

var a = 1;
const obj = {
  a: 2,
  func: function() {
    console.log(this.a);
    const inner = () => console.log(this.a);
    inner();
  }
};
obj.func();
const func = obj.func;
func();
//  2 → 2 → 1 → 1

var length = 10;
function fn() {
  console.log(this.length);
}
var obj = {
  length: 5,
  method: function (fn) {
    fn(); // 输出什么 ？
    arguments[0](); // 输出什么
    fn.call(obj, 12); // 输出什么  ？
  }
};
obj.method(fn, 1, 2);
// 10 3 5
```

### var 和 let 在循环中的作用域

```JavaScript
var a = 10;
function test() {
  var a = 20;
  for (var i = 0; i < 3; i++) {
    let a = 30;
    console.log(`i: ${i}, a: ${a}`);
  }
  console.log(`a: ${a}`);
}
test();
console.log(`a: ${a}`);
```

### this 在闭包中的问题

```JavaScript
var obj = {
  count: 0,
  increment: function() {
    setTimeout(function() {
      this.count++;
      console.log(this.count);
    }, 1000);
  }
};

obj.increment();
```

- 在 increment 方法中，setTimeout 内的回调函数会创建一个新的上下文，因此其中的 this 指向的是 window（或者在严格模式下为 undefined），而不是 obj。 由于 this.count 在 window 上没有定义，或者在严格模式下没有 this，回调函数会报错或者输出 NaN。
